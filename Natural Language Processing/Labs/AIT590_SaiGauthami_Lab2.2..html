def _scrape(url):
    """
    Use BeautifulSoup to scrape the webpage text contents.
    """    
    scraped_textdata = urllib.request.urlopen(url)
    textdata = scraped_textdata.read()
    parsed_textdata = bs.BeautifulSoup(textdata,'lxml')
    paragraphs = parsed_textdata.find_all('p')
    formated_text = ""
    sentences = []

    for para in paragraphs:
        formated_text += para.text
        sentences.append(para.text)
    
    return formated_text,sentences

def co_occurrence_matrix(corpus, ngram):
    vocab = set(corpus)
    vocab = list(vocab)
    vocab_index = {word: i for i, word in enumerate(vocab)}
    ngram_freq_dist = nltk.FreqDist(ngram['ngram']).most_common(ngram.shape[0]) # Frequency distribution of bigrams ((word1, word2), num_occurrences)
    co_occurrence_matrix = np.zeros((len(vocab), len(vocab))) # Initialise co-occurrence matrix                                           
    for ngram in ngram_freq_dist: # Loop through the bigrams taking the current and previous word,
        current = ngram[0][1]          # and the number of occurrences of the bigram.  
        previous = ngram[0][0]
        count = ngram[1]
        pos_current = vocab_index[current]
        pos_previous = vocab_index[previous]
        co_occurrence_matrix[pos_current][pos_previous] = count
    co_occurrence_matrix = np.matrix(co_occurrence_matrix)
 
    # return the matrix and the index
    return co_occurrence_matrix, vocab_index
